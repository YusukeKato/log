NEORT “Random Sin Wave”

作品

JavaScript

p5.js

var offset = 0;
var strum = 1;

function setup() {
  createCanvas(windowWidth, windowHeight);
}

function draw() {
  background(255, 0, 0);
  stroke(4);
  noFill();
  beginShape();
  vertex(0, height);
  for(var x = 0; x < width; x++){
    var angle = offset + x * 0.01;
    var y = map(sin(angle)+random(5), -strum, strum, 150, 250);
    vertex(x, y+random(50));
  }
  vertex(width, height);
  endShape();
  offset += 0.1;
}
